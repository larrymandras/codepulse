```typescript
useEffect(() => {
  if (!runId) return;
  // Reset state for new run
  setLogLines([]);
  setIframeContent("");
  setAccumulatedText("");
  setStreamStatus("streaming");
  setProgress(0);
  setErrorMessage(null);

  const controller = new AbortController();
  let textBuffer = "";

  streamRunEvents(runId, {
    onToken: (text) => {
      textBuffer += text;
      setAccumulatedText(textBuffer);
      setLogLines(prev => [...prev, text]);
      // Extract artifact HTML from accumulated text
      const artifact = extractArtifact(textBuffer);
      if (artifact) setIframeContent(artifact);
      // Approximate progress (rough heuristic based on text length)
      setProgress(prev => Math.min(prev + 0.5, 95));
    },
    onError: (err) => {
      setStreamStatus("error");
      setErrorMessage(err.message || "Generation failed. The daemon returned an error.");
    },
    onDone: () => {
      setStreamStatus("complete");
      setProgress(100);
      onGenerationComplete();
    },
    signal: controller.signal,
  });

  return () => controller.abort();
}, [runId]);
```

**Direction generation helper** — parses daemon run output into 3 directions:
```typescript
function parseDirections(text: string): Array<{ title: string; rationale: string; keywords: string[] }> {
  // The daemon agent responds with structured direction output.
  // Parse JSON array if present, otherwise extract from markdown-style output.
  try {
    // Try JSON parse first (agent may respond with JSON)
    const jsonMatch = /\[[\s\S]*\]/m.exec(text);
    if (jsonMatch) {
      const parsed = JSON.parse(jsonMatch[0]);
      if (Array.isArray(parsed) && parsed.length >= 1) {
        return parsed.slice(0, 3).map((d: any) => ({
          title: String(d.title ?? d.name ?? "Direction"),
          rationale: String(d.rationale ?? d.description ?? ""),
          keywords: Array.isArray(d.keywords) ? d.keywords.map(String) : [],
        }));
      }
    }
  } catch { /* fall through to text parsing */ }

  // Fallback: parse markdown headers as directions
  const sections = text.split(/#{1,3}\s+/).filter(Boolean);
  return sections.slice(0, 3).map((section, i) => {
    const lines = section.trim().split("\n");
    return {
      title: lines[0]?.trim() || `Direction ${i + 1}`,
      rationale: lines.slice(1, 3).join(" ").trim() || "Design direction generated by the agent",
      keywords: [],
    };
  });
}
```

**Direction generation function** — triggers a daemon run to generate directions:
```typescript
async function generateDirections() {
  setDirectionsLoading(true);
  setDirections([]);
  try {
    // Create project first
    const project = await createProject({
      name: `Design ${Date.now()}`,
      skill_id: selectedSkillId ?? undefined,
      design_system_id: selectedDesignSystemId ?? undefined,
    });
    setProjectId(project.id);

    // Fetch available agents
    const agents = await fetchAgents();
    const agent = agents.find(a => a.available) ?? agents[0];
    if (!agent) {
      toast.error("No agents available on the daemon");
      setDirectionsLoading(false);
      return;
    }

    // Start a direction-generation run via daemon
    const directionPrompt = [
      `Generate exactly 3 distinct design directions for this project.`,
      ``,
      `Skill: ${selectedSkillId ?? "general"}`,
      `Design System: ${selectedDesignSystemId ?? "default"}`,
      `Brief: ${brief}`,
      ``,
      `For each direction, provide:`,
      `- A short title (3-5 words)`,
      `- A rationale (1-2 sentences explaining the approach)`,
      `- 3-5 keywords describing the visual style`,
      ``,
      `Respond with a JSON array: [{ "title": "...", "rationale": "...", "keywords": ["..."] }]`,
    ].join("\n");

    const { runId: dirRunId } = await createRun({
      agentId: agent.id,
      message: directionPrompt,
      projectId: project.id,
    });

    // Consume the SSE stream to get direction output
    let accumulated = "";
    const controller = new AbortController();

    streamRunEvents(dirRunId, {
      onToken: (text) => {
        accumulated += text;
      },
      onError: (err) => {
        toast.error(err.message || "Direction generation failed");
        setDirectionsLoading(false);
      },
      onDone: () => {
        const parsed = parseDirections(accumulated);
        if (parsed.length > 0) {
          setDirections(parsed);
        } else {
          // Fallback: provide generic directions if parsing fails
          setDirections([
            { title: "Bold & Modern", rationale: "A contemporary approach with strong visual hierarchy and clean lines.", keywords: ["modern", "bold", "clean"] },
            { title: "Classic & Refined", rationale: "A timeless design with elegant typography and restrained use of color.", keywords: ["classic", "refined", "elegant"] },
            { title: "Playful & Expressive", rationale: "An energetic direction with dynamic layouts and vibrant accents.", keywords: ["playful", "dynamic", "vibrant"] },
          ]);
          toast.info("Using default directions — agent output could not be parsed");
        }
        setDirectionsLoading(false);
      },
      signal: controller.signal,
    });
  } catch (err) {
    toast.error("Failed to generate directions");
    setDirectionsLoading(false);
  }
}
```
