Add a new `GET /api/email-assets` endpoint to `template_routes.py` that lists all assets in the `email-assets` Supabase Storage bucket. Insert this endpoint BEFORE the existing `GET /api/email-assets/{path:path}` route (line 305) so FastAPI matches the list endpoint first.

```python
@router.get("/email-assets")
async def list_email_assets(
    folder: str | None = Query(None, description="Filter by subfolder: avatars or logos"),
) -> list[dict[str, Any]]:
    """List all assets in the email-assets storage bucket."""
    import httpx as _httpx

    _require_db()

    prefix = f"email-assets/{folder}" if folder else "email-assets"
    # Supabase Storage list API: POST /storage/v1/object/list/{bucket}
    list_url = f"{_supabase_url.rstrip('/')}/storage/v1/object/list/email-assets"

    body: dict[str, Any] = {"prefix": folder or "", "limit": 1000}

    async with _httpx.AsyncClient(timeout=10.0) as client:
        resp = await client.post(
            list_url,
            json=body,
            headers={
                "apikey": _supabase_key,
                "Authorization": f"Bearer {_supabase_key}",
                "Content-Type": "application/json",
            },
        )
        if resp.status_code != 200:
            log.warning("email.assets.list_failed", status=resp.status_code, body=resp.text[:200])
            raise HTTPException(status_code=502, detail="Failed to list assets from storage")

    items = resp.json()
    public_base = f"{_supabase_url.rstrip('/')}/storage/v1/object/public/email-assets"
    result = []
    for item in items:
        if item.get("id") is None:
            continue  # Skip folder entries (they have no id)
        name = item.get("name", "")
        folder_prefix = folder or ""
        storage_path = f"email-assets/{folder_prefix}/{name}" if folder_prefix else f"email-assets/{name}"
        result.append({
            "name": name,
            "storage_path": storage_path,
            "public_url": f"{public_base}/{folder_prefix}/{name}" if folder_prefix else f"{public_base}/{name}",
            "size": item.get("metadata", {}).get("size"),
            "created_at": item.get("created_at"),
        })

    log.info("email.assets.listed", count=len(result), folder=folder)
    return result
```

CRITICAL: This endpoint MUST be placed before `@router.get("/email-assets/{path:path}")` in the file, otherwise FastAPI will match the path-parameter route first and never reach the list route.
