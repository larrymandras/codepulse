export interface CommandEntry {
  name: string;           // slash command string, e.g. "/run"
  description: string;    // human-readable summary
  category: string;       // e.g. "core", "skills", "mcp"
  parameters?: {
    name: string;
    type: string;
    required: boolean;
    description?: string;
  }[];
  source?: string;        // which manifest file it came from
}

2. Create `src/hooks/useCommandCatalog.ts`:
- Import `useAstridrWS` from `@/contexts/AstridrWSContext`
- Import `CommandEntry` from `@/types/commands`
- Export `useCommandCatalog()` hook that returns `{ commands: CommandEntry[], status: "loading" | "ready" | "error", error?: string }`
- On mount, call `subscribeEvent("commands.catalog", callback)` to listen for the catalog payload
- The callback expects a message shaped as `{ event_type: "commands.catalog", data: { tools: CommandEntry[], pipes: any[], commands: any[] } }` -- extract the `tools` array (and optionally flatten pipes and commands into the list with appropriate category markers)
- When `status` from `useAstridrWS()` is `"connected"` and no catalog received yet, set internal state to `"loading"`
- When `status` is `"disconnected"`, clear commands array, set internal state to `"error"`, set error to `"Registry unavailable. Connect to Astridhr to load the command catalog."`
- When `status` transitions from `"connected"` to `"reconnecting"` or `"disconnected"`, clear commands and show loading/error accordingly
- Use `useState` for commands array and `useEffect` for subscription lifecycle
- Clean up subscription on unmount via the returned unsubscribe function
- Validate incoming data: `Array.isArray(msg.data?.tools)` before setting state (T-58-01 mitigation)
