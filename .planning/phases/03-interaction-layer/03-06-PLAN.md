```typescript
/**
 * Insights Chat — LLM-powered Q&A over CodePulse operational data.
 *
 * Uses structured tool calls to query Convex data (D-11).
 * Returns GenerativeBlock[] for rendering in InsightsChat page (D-12).
 *
 * Phase 3, Plan 06: IL-06.
 */
import { action } from "./_generated/server";
import { api } from "./_generated/api";
import { v } from "convex/values";

// ─── Tool Definitions ────────────────────────────────────────────────────────

// These are the safe, pre-scoped queries the LLM can call (D-11).
// No raw DB access — only named queries exposed as tools.

type ToolResult = { type: string; [key: string]: unknown };

const TOOL_HANDLERS: Record<string, (ctx: any) => Promise<ToolResult>> = {
  cost_summary: async (ctx) => {
    const metrics = await ctx.runQuery(api.metrics.dashboardSummary);
    return {
      type: "metric",
      label: "Total Cost",
      value: metrics?.totalCost ?? 0,
      trend: "neutral",
    };
  },
  error_counts: async (ctx) => {
    const alerts = await ctx.runQuery(api.alerts.list);
    const errorCount = (alerts ?? []).filter((a: any) => a.severity === "error" || a.type === "error").length;
    return {
      type: "metric",
      label: "Error Count",
      value: errorCount,
      trend: errorCount > 5 ? "up" : "neutral",
    };
  },
  agent_status: async (ctx) => {
    const agents = await ctx.runQuery(api.agents.list);
    const rows = (agents ?? []).map((a: any) => [
      a.name || a.agentName || "Unknown",
      a.status || "unknown",
      a.lastSeen ? new Date(a.lastSeen).toLocaleString() : "N/A",
    ]);
    return {
      type: "table",
      columns: ["Agent", "Status", "Last Seen"],
      rows,
    };
  },
  session_list: async (ctx) => {
    const sessions = await ctx.runQuery(api.sessions.list);
    const rows = (sessions ?? []).slice(0, 20).map((s: any) => [
      s.sessionId ?? s._id,
      s.status ?? "unknown",
      s.agentName ?? "N/A",
    ]);
    return {
      type: "table",
      columns: ["Session", "Status", "Agent"],
      rows,
    };
  },
  alert_summary: async (ctx) => {
    const alerts = await ctx.runQuery(api.alerts.list);
    const total = (alerts ?? []).length;
    const unread = (alerts ?? []).filter((a: any) => !a.read).length;
    return {
      type: "metric",
      label: "Active Alerts",
      value: `${unread} unread / ${total} total`,
      trend: unread > 3 ? "up" : "neutral",
    };
  },
};

// ─── Question → Tool Mapping (keyword-based, no LLM needed for v1) ──────────

// Instead of calling an external LLM (which requires API keys in Convex env),
// use keyword matching to select the right tool. This is the safe, auditable
// path per D-11 — no raw query generation, no prompt injection risk.
// When Ástríðr v4.0 LiteLLM integration is available in Convex, swap this
// for actual LLM tool-calling.

function selectTools(question: string): string[] {
  const q = question.toLowerCase();
  const selected: string[] = [];

  if (q.includes("cost") || q.includes("spend") || q.includes("budget")) selected.push("cost_summary");
  if (q.includes("error") || q.includes("fail") || q.includes("broken")) selected.push("error_counts");
  if (q.includes("agent") || q.includes("bot")) selected.push("agent_status");
  if (q.includes("session") || q.includes("run") || q.includes("execution")) selected.push("session_list");
  if (q.includes("alert") || q.includes("notification") || q.includes("warn")) selected.push("alert_summary");

  // Default: if nothing matched, return a general summary
  if (selected.length === 0) selected.push("cost_summary", "error_counts", "alert_summary");

  return selected;
}

// ─── Action ──────────────────────────────────────────────────────────────────

export const ask = action({
  args: {
    question: v.string(),
  },
  handler: async (ctx, { question }): Promise<ToolResult[]> => {
    const toolNames = selectTools(question);
    const results: ToolResult[] = [];

    for (const name of toolNames) {
      const handler = TOOL_HANDLERS[name];
      if (handler) {
        try {
          const result = await handler(ctx);
          results.push(result);
        } catch {
          results.push({
            type: "markdown",
            content: `Failed to query ${name}.`,
          });
        }
      }
    }

    // Add a summary markdown block at the end
    results.push({
      type: "markdown",
      content: `Queried ${toolNames.length} data source${toolNames.length !== 1 ? "s" : ""} for: "${question}"`,
    });

    return results;
  },
});

// Export helper for testing
export { selectTools, TOOL_HANDLERS };
```

Note: This uses keyword-based tool selection rather than a real LLM call. This is intentional — per RESEARCH.md Open Question #3, the LLM provider and API keys for Convex are unconfirmed. The keyword approach satisfies D-11 (structured tool calls, no raw DB) and D-12 (returns GenerativeBlock-shaped data). When LiteLLM becomes available in Convex env, the `selectTools` function can be replaced with an actual LLM call.

**Step 1: Create `src/pages/InsightsChat.tsx`:**

```typescript
/**
 * InsightsChat — LLM-powered Q&A over CodePulse operational data.
 *
 * Distinct from Agent Chat (which sends tasks to Ástríðr).
 * Reuses ChatBubble + BlockRenderer for consistent rendering (D-12).
 * Backend uses structured Convex tool calls (D-11).
 *
 * Phase 3, Plan 06: IL-06.
 */
import { useState, useRef, useEffect, useCallback } from "react";
import { useAction } from "convex/react";
import { api } from "../../convex/_generated/api";
import { ChatBubble } from "@/components/ChatBubble";
import { ChatInput } from "@/components/ChatInput";
import { Loader2 } from "lucide-react";
import type { ChatMessage, GenerativeBlock } from "@/types/generative-blocks";

function generateId(): string {
  return crypto.randomUUID();
}

export default function InsightsChat() {
  const askInsights = useAction(api.insightsChat.ask);
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const scrollRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom on new messages
  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages, loading]);

  const handleSend = useCallback(async (text: string) => {
    if (!text.trim()) return;

    // Add user message
    const userMsg: ChatMessage = {
      id: generateId(),
      role: "user",
      content: text,
      streaming: false,
      timestamp: Date.now(),
    };
    setMessages((prev) => [...prev, userMsg]);
    setLoading(true);
    setError(null);

    try {
      const blocks = await askInsights({ question: text });
      // Add assistant response with blocks
      const assistantMsg: ChatMessage = {
        id: generateId(),
        role: "assistant",
        blocks: blocks as GenerativeBlock[],
        streaming: false,
        timestamp: Date.now(),
      };
      setMessages((prev) => [...prev, assistantMsg]);
    } catch (err) {
      setError("Query failed. Try rephrasing your question.");
    } finally {
      setLoading(false);
    }
  }, [askInsights]);

  return (
    <div className="flex flex-col h-full">
      {/* Header */}
      <div className="p-4 border-b border-(--border)">
        <h1 className="text-base font-semibold">Insights</h1>
      </div>

      {/* Message area */}
      <div ref={scrollRef} className="flex-1 overflow-y-auto p-4 space-y-4">
        {messages.length === 0 && !loading && (
          <p className="text-(--muted-foreground) text-sm text-center mt-8">
            Ask a question to get started.
          </p>
        )}

        {messages.map((msg) => (
          <ChatBubble
            key={msg.id}
            role={msg.role}
            content={msg.content}
            blocks={msg.blocks}
            streaming={msg.streaming}
            timestamp={msg.timestamp}
          />
        ))}

        {loading && (
          <div className="flex items-center gap-2 text-(--muted-foreground) text-sm">
            <Loader2 className="h-4 w-4 animate-spin" />
            Querying your data...
          </div>
        )}

        {error && (
          <p className="text-(--destructive) text-sm">{error}</p>
        )}
      </div>

      {/* Input */}
      <div className="p-4 border-t border-(--border)">
        <ChatInput
          onSend={handleSend}
          disabled={loading}
          placeholder="Ask about cost, errors, sessions..."
        />
      </div>
    </div>
  );
}
```
