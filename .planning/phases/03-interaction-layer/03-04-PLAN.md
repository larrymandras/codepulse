2. Add useEffect for keyboard handler:
```typescript
useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    // Guard: skip if user is typing in an input or textarea
    if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return;

    const items = filteredItems; // the currently filtered & sorted InboxItem[]
    if (!items.length) return;

    if (e.key === "ArrowDown") {
      e.preventDefault();
      setFocusedIndex((prev) => (prev === null ? 0 : Math.min(prev + 1, items.length - 1)));
    }
    if (e.key === "ArrowUp") {
      e.preventDefault();
      setFocusedIndex((prev) => (prev === null ? 0 : Math.max(prev - 1, 0)));
    }
    if (e.key === "Enter" && focusedIndex !== null) {
      e.preventDefault();
      const item = items[focusedIndex];
      setExpandedId((prev) => (prev === item.id ? null : item.id));
    }
    if (e.key === "Escape") {
      setFocusedIndex(null);
    }
    if (e.key === "a" && focusedIndex !== null) {
      const item = items[focusedIndex];
      if (item.type === "approval") {
        e.preventDefault();
        handleApprove(item.id);
      }
    }
    if (e.key === "r" && focusedIndex !== null) {
      const item = items[focusedIndex];
      if (item.type === "approval") {
        e.preventDefault();
        handleReject(item.id);
      }
    }
  };
  document.addEventListener("keydown", handler);
  return () => document.removeEventListener("keydown", handler);
}, [focusedIndex, filteredItems]);
```
Note: `filteredItems` must be a stable reference — if it's computed inline, extract to a `useMemo`.
