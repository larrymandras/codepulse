2. Add a `groupIntoRounds` function:
```typescript
function groupIntoRounds(blocks: Array<{ type: string; [key: string]: unknown }>, streaming: boolean): Round[] {
  const rounds: Round[] = [];
  let current: Round | null = null;

  for (const block of blocks) {
    // A "thinking" or "reasoning" block starts a new round
    if (block.type === "thinking" || block.type === "reasoning") {
      if (current) {
        current.done = true;
        rounds.push(current);
      }
      current = { index: rounds.length + 1, blocks: [block], done: false };
    } else {
      if (!current) {
        // First block is not thinking — create round 1
        current = { index: 1, blocks: [], done: false };
      }
      current.blocks.push(block);
    }
  }

  if (current) {
    // Last round: done if NOT streaming, active if streaming
    current.done = !streaming;
    rounds.push(current);
  }

  return rounds;
}
```

3. Rewrite the component to render rounds as `<details>` accordion:
```typescript
export function RunTimeline({ blocks, streaming = false }: RunTimelineProps) {
  const showThinking = streaming && blocks.length === 0;
  const rounds = useMemo(() => groupIntoRounds(blocks, streaming), [blocks, streaming]);

  if (showThinking) {
    return <div className="text-(--muted-foreground) text-sm animate-pulse">Thinking...</div>;
  }

  return (
    <div className="flex flex-col gap-2">
      {rounds.map((round) => {
        const toolCallCount = round.blocks.filter(b => b.type === "tool_use").length;
        const isActive = !round.done;

        return (
          <details key={round.index} open={isActive} className="group">
            <summary
              className={`flex items-center gap-2 cursor-pointer select-none p-2 text-sm font-semibold ${
                isActive ? "border-l-4 border-(--status-warn)" : "border-l-4 border-transparent"
              }`}
            >
              <span>Round {round.index}</span>
              <span className="text-xs text-(--muted-foreground) font-normal">
                {toolCallCount} tool call{toolCallCount !== 1 ? "s" : ""}
              </span>
              {round.done && <span className="text-xs text-(--status-ok)">✓</span>}
              {isActive && streaming && (
                <span className="h-2 w-2 rounded-full bg-(--status-warn) animate-[live-update-pulse_2s_ease-in-out_infinite]" />
              )}
            </summary>
            <div className="pl-4 flex flex-col gap-1">
              {round.blocks.map((block, idx) => (
                <RunBlock key={idx} block={block} />
              ))}
            </div>
          </details>
        );
      })}
    </div>
  );
}
```

3. Add `buildFlowGraph` function (memoized):
```typescript
const flowGraph = useMemo(() => {
  if (activeTab !== "flow") return { nodes: [], edges: [] };
  
  const g = new dagre.graphlib.Graph();
  g.setDefaultEdgeLabel(() => ({}));
  g.setGraph({ rankdir: "TB", nodesep: 40, ranksep: 60 });

  const nodes: Node[] = [];
  const edges: Edge[] = [];
  const toolCallIds = new Map<string, string>();

  blocks.forEach((block: any, i: number) => {
    if (block.type === "tool_use") {
      const id = `tool_use_${i}`;
      toolCallIds.set(block.tool_call_id ?? `tc_${i}`, id);
      g.setNode(id, { width: 160, height: 40 });
      nodes.push({
        id,
        type: "default",
        data: { label: block.name ?? block.tool_name ?? "Tool" },
        position: { x: 0, y: 0 },
        style: {
          background: "var(--secondary)",
          border: "1px solid var(--border)",
          color: "var(--foreground)",
          fontSize: "12px",
          borderRadius: "0px",  // --radius: 0
        },
      });
    }
    if (block.type === "tool_result") {
      const sourceId = toolCallIds.get(block.tool_call_id ?? `tc_${i}`);
      const id = `tool_result_${i}`;
      g.setNode(id, { width: 160, height: 40 });
      if (sourceId) {
        g.setEdge(sourceId, id);
        edges.push({
          id: `e_${sourceId}_${id}`,
          source: sourceId,
          target: id,
          style: { stroke: "var(--muted-foreground)" },
        });
      }
      nodes.push({
        id,
        type: "default",
        data: { label: "Result" },
        position: { x: 0, y: 0 },
        style: {
          background: "var(--secondary)",
          border: "1px solid var(--border)",
          color: "var(--foreground)",
          fontSize: "12px",
          borderRadius: "0px",
        },
      });
    }
  });

  if (nodes.length > 0) {
    dagre.layout(g);
    nodes.forEach((n) => {
      const nodeWithPos = g.node(n.id);
      if (nodeWithPos) {
        n.position = { x: nodeWithPos.x - 80, y: nodeWithPos.y - 20 };
      }
    });
  }

  return { nodes, edges };
}, [blocks, activeTab]);
```
