**File 1: `convex/agents.ts`** — add after existing `listAll`:

```typescript
import { paginationOptsValidator } from "convex/server";

export const listAllPaginated = query({
  args: { paginationOpts: paginationOptsValidator },
  handler: async (ctx, args) => {
    return await ctx.db
      .query("agents")
      .order("desc")
      .paginate(args.paginationOpts);
  },
});
```

Note: The `agents` table is NOT a high-volume archival target (per D-06), so no archived filter. Check if the agents table has a `by_timestamp` or `by_creation_time` index; if not, use `.order("desc")` without `.withIndex()` (Convex allows paginating on the default `_creationTime` order).
