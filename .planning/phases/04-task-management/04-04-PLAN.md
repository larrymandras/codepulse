    **2. Hot-reload status bar (D-10):**
    Add state:
    ```typescript
    const [reloadStatus, setReloadStatus] = useState<"pending" | "validating" | "applied" | "confirmed" | "error" | null>(null);
    const [reloadError, setReloadError] = useState<string | undefined>(undefined);
    ```

    Modify the existing apply handler to update status through the lifecycle:
    ```typescript
    async function handleApply() {
      setReloadStatus("pending");
      try {
        // Dry-run validation first
        setReloadStatus("validating");
        const dryResult = await sendCommand({ type: "config.update", section, changes: yamlContent, dry_run: true });
        if (dryResult.status === "error") {
          setReloadStatus("error");
          setReloadError(`Validation error: ${dryResult.error}`);
          return;
        }
        // Actual apply
        setReloadStatus("pending");
        const result = await sendCommand({ type: "config.update", section, changes: yamlContent, dry_run: false });
        if (result.status === "ok") {
          setReloadStatus("applied");
          setOriginalContent(yamlContent);
          setIsDirty(false);
          // Wait for WS ack confirmation (simulate with timeout if no explicit confirmed event)
          setTimeout(() => setReloadStatus("confirmed"), 1500);
          setTimeout(() => setReloadStatus(null), 4000);
        } else {
          setReloadStatus("error");
          setReloadError(result.error ?? "Unknown error");
        }
      } catch (err) {
        setReloadStatus("error");
        setReloadError(err instanceof Error ? err.message : "Command failed");
      }
    }
    ```

    Render HotReloadBar below the Apply button:
    ```tsx
    <HotReloadBar status={reloadStatus} errorMessage={reloadError} />
    ```

    Disable Apply button when reloadStatus is "pending" or "validating":
