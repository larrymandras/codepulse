```typescript
import { useReducer, useEffect, useMemo } from "react";
import { useAstridrWS, type WSStatus } from "@/contexts/AstridrWSContext";

export type LiveStateSlice = {
  agentStatus: "idle" | "running" | "paused" | null;
  activeRunId: string | null;
  activeRunProgress: number | null;
  liveMetricDeltas: Record<string, number>;
  connectionHealth: WSStatus;
};

type LiveStateAction =
  | { type: "SET_AGENT_STATUS"; payload: LiveStateSlice["agentStatus"] }
  | { type: "SET_ACTIVE_RUN"; payload: { id: string | null; progress: number | null } }
  | { type: "SET_METRIC_DELTA"; payload: { key: string; value: number } }
  | { type: "SET_CONNECTION_HEALTH"; payload: WSStatus }
  | { type: "CLEAR_ALL" };

const INITIAL_STATE: LiveStateSlice = {
  agentStatus: null,
  activeRunId: null,
  activeRunProgress: null,
  liveMetricDeltas: {},
  connectionHealth: "disconnected",
};

function liveStateReducer(state: LiveStateSlice, action: LiveStateAction): LiveStateSlice {
  switch (action.type) {
    case "SET_AGENT_STATUS":
      return { ...state, agentStatus: action.payload };
    case "SET_ACTIVE_RUN":
      return { ...state, activeRunId: action.payload.id, activeRunProgress: action.payload.progress };
    case "SET_METRIC_DELTA":
      return { ...state, liveMetricDeltas: { ...state.liveMetricDeltas, [action.payload.key]: action.payload.value } };
    case "SET_CONNECTION_HEALTH":
      return { ...state, connectionHealth: action.payload };
    case "CLEAR_ALL":
      return { ...INITIAL_STATE, connectionHealth: state.connectionHealth };
    default:
      return state;
  }
}
```

**Ping-based latency measurement (per resolved RESEARCH.md open question):**
- When status is "connected", send a periodic ping every 30 seconds via `sendCommand({ action: "ping" })` (note: this sends `{action: "ping", request_id: "..."}` which the Astrid backend handles via the `ping` action in ws_telemetry.py, returning an ack)
- Measure RTT: `const start = performance.now()`, await the ack, `const rtt = Math.round(performance.now() - start)`
- Store in `latencyMs` state
- On disconnect, clear latencyMs to null
- Do NOT send pings when disconnected (guard with wsStatus check)
- Use `useRef` for the interval timer, clear on unmount and on disconnect

**IMPORTANT:** The `sendCommand` function sends JSON with the fields from the cmd object plus a generated `request_id`. The backend ws_telemetry.py dispatches messages with a `type` field to the command dispatcher. Messages with only `action` (no `type`) are handled directly by ws_telemetry.py's action handlers (subscribe/unsubscribe). Since `ping` is not currently a recognized action, the message will be silently ignored. Therefore, use `sendCommand({ type: "ping" })` instead — this routes through the CommandDispatcher. If the dispatcher has no handler for "ping", it returns an error ack, but the RTT is still measurable from the ack round-trip time. The latency value is the network RTT regardless of ack status.

Uptime display: calculate from `connectedAt` using an interval timer (update every second). Format: "Connected {N}m {N}s ago" or "--" when disconnected.
