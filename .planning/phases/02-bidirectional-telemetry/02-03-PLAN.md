**Step 1: Create `src/hooks/useLiveFlash.ts` — reusable flash animation hook.**

```typescript
import { useRef, useCallback } from "react";

/**
 * useLiveFlash — triggers the .live-update-flash CSS animation
 * on an element ref, debounced to 1 flash per second max.
 *
 * Usage:
 *   const { flashRef, triggerFlash } = useLiveFlash();
 *   <div ref={flashRef}>...</div>
 *   // In WS callback: triggerFlash()
 */
export function useLiveFlash<T extends HTMLElement = HTMLDivElement>() {
  const flashRef = useRef<T>(null);
  const lastFlashRef = useRef(0);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const triggerFlash = useCallback(() => {
    const el = flashRef.current;
    if (!el) return;
    const now = Date.now();
    if (now - lastFlashRef.current < 1000) return; // debounce: 1s
    lastFlashRef.current = now;
    el.classList.remove("live-update-flash"); // reset if still animating
    // Force reflow to restart animation
    void el.offsetWidth;
    el.classList.add("live-update-flash");
    if (timerRef.current) clearTimeout(timerRef.current);
    timerRef.current = setTimeout(() => {
      el?.classList.remove("live-update-flash");
    }, 620);
  }, []);

  return { flashRef, triggerFlash };
}
```
