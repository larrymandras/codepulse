    **Multi-select handlers (per D-06):**
    ```typescript
    function handleSelect(id: string, checked: boolean) {
      setSelectedIds(prev => {
        const next = new Set(prev);
        checked ? next.add(id) : next.delete(id);
        return next;
      });
    }

    async function handleBulkConvert() {
      const selected = findings.filter(f => selectedIds.has(f._id));
      for (const finding of selected) {
        const defaults = findingToTaskDefaults(finding);
        const taskId = await createTask({ ...defaults, title: defaults.title ?? finding.description, priority: defaults.priority ?? "medium" });
        await linkTask({ id: finding._id, taskId: taskId });
      }
      setSelectedIds(new Set());
      toast.success(`Converted ${selected.length} findings to tasks`);
    }
    ```

    **Inline conversion handler (per D-06):**
    ```typescript
    function handleCreateTask(finding: ...) {
      const defaults = findingToTaskDefaults(finding);
      setPrefillData(defaults);
      setShowCreateForm(true);
    }
    ```

    ```typescript
    async function handleAcknowledge(id: string) {
      await updateStatus({ id, status: "acknowledged" });
    }

      await updateStatus({ id, status: "dismissed" });
    }
    ```
